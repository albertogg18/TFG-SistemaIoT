#include <Arduino.h>
#include <DHT.h>
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>

// --- Configuración Wi-Fi (Router Virtual de Wokwi) ---
const char* ssid = "MI_WIFI";
const char* password = "MI_PASSWORD";

// --- Configuración del servidor al que enviaremos los datos ---
const char* serverName = "http://<DIRECCION>:3000/api/sensores";

// Configuracion del sensor DHT22 (Temperatura y Humedad )
#define DHTPIN 15     
#define DHTTYPE DHT22

// Definición del PIN para el sensor de humedad del suelo
#define SOIL_MOISTURE_PIN 34

// Definición del PIN para el sensor de luz
#define LDR_PIN 32

// ---- Creamos el objeto de los sensores ----

// Creacion sensor de temperatura y humedad DHT22
DHT dht(DHTPIN, DHTTYPE);

// ---- Inicializacion de sensores consola ----
void setup() {
  // Iniciamos la consola serie a 115200 baudios
  Serial.begin(115200);
  
  // Pequeña pausa para que la consola arranque limpia
  delay(1000);

  Serial.println("---- Sistema IoT - Lectura de Sensores ----");

  Serial.println("Conectando a Wi-Fi...");
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("Conectado a Wi-Fi!");
  Serial.print("Dirección IP: ");
  Serial.println(WiFi.localIP());

  // Arrancamos el sensor DHT22
  dht.begin();

  // Configuramos SOIL_MOISTURE_PIN como entrada
  pinMode(SOIL_MOISTURE_PIN, INPUT);

  // Configuramos LDR_PIN como entrada
  pinMode(LDR_PIN, INPUT);
}

// ---- Loop principal ----
void loop() {
  // Pausa de 10 segundos entre lecturas
  delay(10000);

  // ---- Lecturas de sensores ----

  // Leemos la humedad y la temperatura
  float humedad = dht.readHumidity();
  float temperatura = dht.readTemperature();

  // Leemos el nivel de luz
  // Mapeamos el estado del LDR a un valor de luminosidad
  int ldrEstado = analogRead(LDR_PIN);
  float luminosidad = map(ldrEstado, 0, 4095, 0, 1000); // Ajustar según calibración

  // Leemos el nivel de humedad del suelo
  // Mapeamos el valor a un porcentaje (ajustar según calibración)
  int humedadSueloValor = analogRead(SOIL_MOISTURE_PIN);
  int humedadSueloPorcentaje = map(humedadSueloValor, 4095, 0, 0, 100);

  // ---- Comprobaciones de errores ----
  /*
  // Comprobacion en el sensor DHT22
  if (isnan(humedad) || isnan(temperatura)) {
    Serial.println("¡Error: No se pudo leer el sensor DHT22!");
    return; // Volvemos al inicio del loop para intentar de nuevo
  }

  // Comprobacion en el sensor BH1750
  if(luminosidad < 0) {
    Serial.println("¡Error: No se pudo leer el sensor BH1750!");
    return; // Volvemos al inicio del loop para intentar de nuevo
  }

  // Comprobacion en el sensor de humedad del suelo
  if(humedadSueloValor < 0 || humedadSueloValor > 4095) {
    Serial.println("¡Error: No se pudo leer el sensor de humedad del suelo!");
    return; // Volvemos al inicio del loop para intentar de nuevo
  }*/

  // --- Empaquetado de datos en formato JSON ---
  JsonDocument doc;
  doc["dispositivo_id"] = "ESP32_0X"; // ID dispositivo
  
  JsonObject sensores = doc["sensores"].to<JsonObject>();
  sensores["temperatura"] = temperatura; // Temperatura en °C
  sensores["humedad_aire"] = humedad; // Humedad en %
  sensores["humedad_suelo"] = humedadSueloPorcentaje; // Humedad del suelo en %
  sensores["luminosidad"] = luminosidad; // Nivel de luz en lux
  
  doc["evento_riego"] = false; // Evento de riego (falso por ahora)

  String jsonOutput;
  serializeJson(doc, jsonOutput);

  // --- Envío de datos al servidor ---
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/json");

    int httpResponseCode = http.POST(jsonOutput);

    if (httpResponseCode > 0) {
      String response = http.getString();
      Serial.print("Respuesta del servidor: ");
      Serial.println(response);
    } else {
      Serial.print("Error en la solicitud HTTP: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  } else {
    Serial.println("¡Error: No se está conectado a Wi-Fi!");
  }
}
